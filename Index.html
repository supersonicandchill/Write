// ============== TAILWIND INIT ==============
function initTailwind() {
tailwind.config = { content: ["./**/*.{html,js}"], theme: { extend: {} } }
}

// ============== DATA MODEL ==============
let bookData = {
title: "Your Kid is the Worst: How Holding Their Hand May Be Holding Them Back",
totalWordsGoal: 17500,
chapters: [],
dailyLog: {}, // date -> { minutes: number, words: number }
streak: 0,
lastDate: null
}

let currentTimer = {
timeLeft: 25 * 60,
isRunning: false,
isBreak: false,
interval: null,
currentChapterId: null
}

let audioContext;

// ============== CHAPTER HELPERS ==============
function createChapter(title) {
return {
id: 'ch-' + Date.now(),
title: title || `Chapter ${bookData.chapters.length + 1}`,
targetWords: Math.round(bookData.totalWordsGoal / Math.max(bookData.chapters.length + 1, 1)),
wordsWritten: 0,
status: 'Not Started',
content: '',
milestones: { outline: false, draft: false, edit: false }
}
}

function saveData() {
localStorage.setItem('focusScriptData', JSON.stringify(bookData))
}

function loadData() {
const saved = localStorage.getItem('focusScriptData')
if (saved) bookData = JSON.parse(saved)
else {
// First time setup
for (let i = 1; i <= 8; i++) {
bookData.chapters.push(createChapter(`Chapter ${i}: The Myth of Constant Supervision`))
}
saveData()
}
updateGlobalProgress()
}

function updateGlobalProgress() {
const totalWritten = bookData.chapters.reduce((sum, ch) => sum + ch.wordsWritten, 0)
const completedChapters = bookData.chapters.filter(ch => ch.status === 'Done').length
// Update dashboard ring etc.
document.getElementById('total-words').textContent = totalWritten
const percent = Math.min(Math.round((totalWritten / bookData.totalWordsGoal) * 100), 100)
const circumference = 2 * Math.PI * 88
const offset = circumference - (circumference * percent / 100)
const circle = document.getElementById('progress-circle')
if (circle) circle.setAttribute('stroke-dashoffset', offset)
document.getElementById('chapter-progress').textContent = `${completedChapters} of ${bookData.chapters.length} chapters`
}

// ============== RENDER CHAPTER LIST ==============
function renderChapters() {
const container = document.getElementById('chapter-list')
container.innerHTML = ''
bookData.chapters.forEach((ch, index) => {
const el = document.createElement('div')
el.className = 'bg-zinc-900 rounded-3xl p-6 cursor-grab active:cursor-grabbing border border-zinc-700 hover:border-emerald-500 transition-colors'
el.draggable = true
el.ondragstart = (e) => e.dataTransfer.setData('text/plain', index)
el.ondragover = (e) => e.preventDefault()
el.ondrop = (e) => {
e.preventDefault()
const from = parseInt(e.dataTransfer.getData('text/plain'))
const temp = bookData.chapters[from]
bookData.chapters[from] = bookData.chapters[index]
bookData.chapters[index] = temp
saveData()
renderChapters()
updateGlobalProgress()
}
el.innerHTML = `
<div class="flex justify-between items-start">
<input type="text" value="${ch.title}" onchange="bookData.chapters[${index}].title=this.value;saveData()" class="bg-transparent font-medium text-lg outline-none flex-1">
<span onclick="deleteChapter(${index});event.stopImmediatePropagation()" class="text-red-400 text-xl leading-none cursor-pointer">×</span>
</div>
<div class="text-xs text-zinc-400 mt-3">${ch.wordsWritten} / ${ch.targetWords} words</div>
<div class="flex gap-2 mt-4">
<div onclick="setChapterStatus(${index},'Not Started');event.stopImmediatePropagation()" class="text-[10px] px-3 py-1 rounded-2xl ${ch.status==='Not Started'?'bg-zinc-700':'bg-zinc-800'}">Not Started</div>
<div onclick="setChapterStatus(${index},'Drafting');event.stopImmediatePropagation()" class="text-[10px] px-3 py-1 rounded-2xl ${ch.status==='Drafting'?'bg-emerald-600':'bg-zinc-800'}">Drafting</div>
<div onclick="setChapterStatus(${index},'Done');event.stopImmediatePropagation()" class="text-[10px] px-3 py-1 rounded-2xl ${ch.status==='Done'?'bg-emerald-500':'bg-zinc-800'}">Done</div>
</div>
<button onclick="openChapterInStudio(${index});event.stopImmediatePropagation()" class="mt-6 w-full py-3 text-sm bg-emerald-500 hover:bg-emerald-600 rounded-3xl">Write now</button>
`
container.appendChild(el)
})
}

function addChapter() {
const ch = createChapter()
bookData.chapters.push(ch)
saveData()
renderChapters()
updateGlobalProgress()
}

function deleteChapter(index) {
if (confirm('Delete this chapter?')) {
bookData.chapters.splice(index, 1)
saveData()
renderChapters()
updateGlobalProgress()
}
}

function setChapterStatus(index, status) {
bookData.chapters[index].status = status
saveData()
renderChapters()
updateGlobalProgress()
}

function openChapterInStudio(index) {
switchTab(3)
const select = document.getElementById('chapter-select')
select.value = index
loadChapterContent()
}

function generateSkeleton() {
if (bookData.chapters.length > 0 && !confirm('Replace current chapters with new skeleton?')) return
bookData.chapters = []
const starters = [
"The Myth of Constant Supervision",
"Why Independence Feels Scary (to Parents)",
"The Hidden Cost of Helicopter Parenting",
"Letting Go Without Losing Control",
"Building Real Resilience in Kids",
"What the Research Actually Shows",
"Your Own Childhood Story Matters",
"The Freedom on the Other Side"
]
for (let i = 0; i < 8; i++) {
const ch = createChapter(`Chapter ${i+1}: ${starters[i]}`)
bookData.chapters.push(ch)
}
saveData()
renderChapters()
updateGlobalProgress()
}

// ============== POMODORO ==============
function initAudio() {
audioContext = new (window.AudioContext || window.webkitAudioContext)()
}

function playTick() {
if (!audioContext) return
const oscillator = audioContext.createOscillator()
oscillator.type = 'sine'
oscillator.frequency.setValueAtTime(440, audioContext.currentTime)
const gain = audioContext.createGain()
gain.gain.value = 0.08
oscillator.connect(gain).connect(audioContext.destination)
oscillator.start()
setTimeout(() => oscillator.stop(), 80)
}

function playPing() {
if (!audioContext) return
const notes = [880, 1100, 1320]
let delay = 0
notes.forEach(freq => {
setTimeout(() => {
const o = audioContext.createOscillator()
o.type = 'sine'
o.frequency.value = freq
const g = audioContext.createGain()
g.gain.value = 0.3
o.connect(g).connect(audioContext.destination)
o.start()
setTimeout(() => o.stop(), 300)
}, delay)
delay += 80
})
}

function startPomodoro() {
if (currentTimer.interval) return
currentTimer.isRunning = true
currentTimer.timeLeft = currentTimer.isBreak ? 5*60 : 25*60
document.getElementById('timer-status').innerHTML = '⏸️'
currentTimer.interval = setInterval(() => {
currentTimer.timeLeft--
document.getElementById('timer-display').textContent =
`${Math.floor(currentTimer.timeLeft/60)}:${(currentTimer.timeLeft%60).toString().padStart(2,'0')}`
if (currentTimer.timeLeft % 2 === 0) playTick()
if (currentTimer.timeLeft <= 0) {
clearInterval(currentTimer.interval)
currentTimer.interval = null
playPing()
finishPomodoro()
}
}, 1000)
}

function toggleTimer() {
if (currentTimer.isRunning) {
clearInterval(currentTimer.interval)
currentTimer.isRunning = false
document.getElementById('timer-status').innerHTML = '▶️'
} else {
startPomodoro()
}
}

function finishPomodoro() {
// Log time
const today = new Date().toISOString().slice(0,10)
if (!bookData.dailyLog[today]) bookData.dailyLog[today] = {minutes:0, words:0}
bookData.dailyLog[today].minutes += 25
saveData()
updateDailyChecklist()
// Celebrate
if (!currentTimer.isBreak) {
alert("🎉 Pomodoro complete! Great work. Your 30-minute promise is closer!")
}
currentTimer.isBreak = !currentTimer.isBreak
document.getElementById('mode-btn').textContent = currentTimer.isBreak ? 'Work (25:00)' : 'Break (5:00)'
if (currentTimer.isBreak) document.getElementById('timer-display').textContent = '05:00'
else document.getElementById('timer-display').textContent = '25:00'
}

function resetPomodoro() {
clearInterval(currentTimer.interval)
currentTimer.isRunning = false
currentTimer.timeLeft = currentTimer.isBreak ? 5*60 : 25*60
document.getElementById('timer-display').textContent = currentTimer.isBreak ? '05:00' : '25:00'
document.getElementById('timer-status').innerHTML = '▶️'
}

function switchMode() {
currentTimer.isBreak = !currentTimer.isBreak
resetPomodoro()
}

function startPomodoroFromDashboard() {
switchTab(2)
startPomodoro()
}

// ============== DAILY CHECKLIST ==============
function updateDailyChecklist() {
const container = document.getElementById('daily-checklist')
const today = new Date().toISOString().slice(0,10)
const log = bookData.dailyLog[today] || {minutes:0}
const minutesDone = log.minutes || 0

container.innerHTML = `
<div class="flex items-center justify-between text-sm ${minutesDone >= 30 ? 'line-through text-emerald-400' : ''}">
<span>✅ 30 minutes writing</span>
<span class="font-mono">${minutesDone} min</span>
</div>
<div onclick="switchTab(3)" class="text-xs bg-zinc-800 hover:bg-emerald-900/30 px-4 py-3 rounded-3xl cursor-pointer mt-3">Review yesterday's writing →</div>
<div onclick="switchTab(1)" class="text-xs bg-zinc-800 hover:bg-emerald-900/30 px-4 py-3 rounded-3xl cursor-pointer mt-2">Mark a chapter milestone</div>
`
}

// ============== STUDIO ==============
function populateChapterSelect() {
const select = document.getElementById('chapter-select')
select.innerHTML = ''
bookData.chapters.forEach((ch, i) => {
const opt = document.createElement('option')
opt.value = i
opt.textContent = ch.title
select.appendChild(opt)
})
}

function loadChapterContent() {
const idx = parseInt(document.getElementById('chapter-select').value)
if (idx < 0 || !bookData.chapters[idx]) return
const ch = bookData.chapters[idx]
document.getElementById('editor').innerHTML = ch.content || '<p>Start writing here…</p>'
document.getElementById('current-chapter-title').textContent = ch.title
// Live word count
const editor = document.getElementById('editor')
editor.oninput = () => {
const text = editor.innerText || ''
const words = text.trim().split(/\s+/).filter(Boolean).length
document.getElementById('word-count').textContent = words
ch.wordsWritten = words
ch.content = editor.innerHTML
saveData()
updateGlobalProgress()
}
}

function toggleFocusMode() {
const container = document.getElementById('editor-container')
container.classList.toggle('fixed')
container.classList.toggle('inset-0')
container.classList.toggle('z-[9999]')
container.classList.toggle('p-12')
if (container.classList.contains('fixed')) {
alert('🔒 Focus Mode ON — Pomodoro still runs in background. Tap anywhere to exit.')
}
}

function exportBook() {
let md = `# ${bookData.title}\n\n`
bookData.chapters.forEach(ch => {
md += `## ${ch.title}\n\n${ch.content ? ch.content.replace(/<[^>]+>/g, '') : 'No content yet'}\n\n`
})
const blob = new Blob([md], {type: 'text/markdown'})
const url = URL.createObjectURL(blob)
const a = document.createElement('a')
a.href = url
a.download = 'your-kid-is-the-worst.md'
a.click()
}

// ============== TAB SWITCHING ==============
function switchTab(n) {
document.querySelectorAll('.tab-content').forEach(t => t.classList.add('hidden'))
document.getElementById('tab-' + n).classList.remove('hidden')
if (n === 1) renderChapters()
if (n === 3) { populateChapterSelect(); loadChapterContent() }
if (n === 4) renderProgress()
}

function renderProgress() {
// Simple canvas word chart
const canvas = document.getElementById('wordChart')
const ctx = canvas.getContext('2d')
ctx.clearRect(0,0,canvas.width,canvas.height)
// Placeholder chart (you can expand)
ctx.fillStyle = '#10b981'
ctx.fillRect(40, 220, 60, 60)
ctx.fillRect(120, 180, 60, 100)
ctx.fillRect(200, 140, 60, 140)
ctx.fillRect(280, 90, 60, 190)
ctx.fillStyle = '#fff'
ctx.font = '12px Inter'
ctx.fillText('Daily words', 20, 30)
}

// ============== MISC ==============

alert(`🔥 Current streak: ${bookData.streak} days!\n\nKeep writing every day to grow it.`)
}

document.getElementById('settings-modal').classList.remove('hidden')
// You can expand settings later
}

document.getElementById('settings-modal').classList.add('hidden')
}

window.onload = function() {
initTailwind()

// Default to Dashboard

console.log('%cFocusScript ready — you’ve got this!', 'color:#10b981; font-size:13px')
}
